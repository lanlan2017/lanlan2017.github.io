/**
 * 整数反转
 * @param x 一个整数
 * @return 反转后的整数
 */
public int reverse(int x) {
    long result = 0;
    int remainder = 0;
    while (x != 0) {
        // 取下最低位
        remainder = x % 10;
        // 移除最低位
        x = x / 10;
        // 前面的最低位乘以10,向前进位，然后把摘下的最低位接到末尾
        result = result * 10 + remainder;
    }
    return result > Integer.MAX_VALUE || result < Integer.MIN_VALUE ? 0 : (int) result;
}
