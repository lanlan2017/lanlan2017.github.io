/**
 * 判断一个数是不是丑数。
 * 只包含质因数2，3，5的数称作丑数
 * 
 * @param n 一个整数
 * @return 如果这个整数n是丑数的话，返回true，否则返回false。
 */
private static boolean isUglyNumber(int n) {
	// 分别去掉2，3，5这些质因数
	while (n % 2 == 0) {
		n = n / 2;
	}
	while (n % 3 == 0) {
		n = n / 3;
	}
	while (n % 5 == 0) {
		n = n / 5;
	}
	// 如果最后的结果是1的话，那就是丑数
	return n == 1 ? true : false;
}
